get steep
$(
y = kx + d
  A1.y = k*A1.x + d
  A2.y = k*A2.x + d

  A1.y - k*A1.x = A2.y - k*A2.x
  A1.y - A2.y = k*A1.x  - k*A2.x
  A1.y - A2.y = k*(A1.x  - A2.x)
  k = (A1.y - A2.y)/(A1.x  - A2.x)

  d = A1.y - k*A1.x
)

get intersection coords
$(
Akx+Ad = Bkx+Bd 
  Akx - Bkx = Bd - Ad
  x (Ak - Bk) = Bd - Ad
  x = (Bd - Ad)/ (Ak - Bk)
)
